### Solution:

```cpp
class Solution {
public:
    vector<string> generateParenthesis(int n) {
	    vector<string> ans;
	    dfs(n, 0, 0, "", ans);
	    return ans;
    }

	void dfs(int n, int left, int right, string s, vector<string>& ans) {
		if (left == right && left == n) {
			ans.push_back(s);
			return;
		}

		if (left < n) {
			dfs(n, left+1, right, s + "(", ans);
		}
		if (left > right) {
			dfs(n, left, right+1, s + ")", ans);
		}
	}
};
```
