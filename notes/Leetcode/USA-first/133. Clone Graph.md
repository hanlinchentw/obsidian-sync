class Solution {
public:
	Node* cloneGraph(Node* node) {
		if (node == NULL)
			return node;
		unordered_map<int, Node*> nodes;
		Node* newNode = new Node(node->val);
		dfs(nodes, newNode, node);
		return newNode;
	}

	void dfs(unordered_map<int, Node*> map, Node* node, Node* original) {
		if (node == NULL || map.find(node->val) != map.end())
			return;
		map[node->val] = node;
		vector<Node*> neighbors = original->neighbors;
		for (int i = 0; i < neighbors.size(); i++) {
			Node* neighbor = neighbors[i];
			Node* newNeighbor = new Node(neighbor->val);
			node->neighbors.push_back(newNeighbor);
			dfs(map, newNeighbor, neighbor);
		}		
	}
};
