class Solution {
public:
    vector<string> findWords(vector<vector<char>>& board, vector<string>& words) {
	    unordered_set<string> ans;
	    for (int i = 0; i < board.size(); i++) {
			for (int j = 0; j < board[0].size(); j++) {
				string s(1, board[i][j]);
				if (canProceed(s, words)) {
					solve(board, s, i, j, ans, words);
				}
			}
		}
	    return vector<string>(ans.begin(), ans.end());
    }

	void solve(const vector<vector<char>>& board, string s, int row, int col, unordered_set<string>& ans, const vector<string>& words) {
		if (0 > row || row >= board.size() || 0 > col || col >= board[0].size() || board[row][col] == '#')) {
			return;
		}
		if (find(words.begin(), words.end(), s) != words.end()) {
			ans.insert(s);
		}
		visited.insert({row, col});
		vector<pair<int, int>> dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
		for (auto dir : dirs) {
			int i = row + dir.first;
			int j = col + dir.second;
			if ((0 <= i && i < board.size()) && (0 <= j && j < board[0].size())) {
				char ch = board[i][j];
				if (ch == '#') continue;
				s.push_back(ch);
				board[i][j] = '#';
				if (canProceed(s, words)) {
					solve(board, s, i, j, ans, words, visited);
				}
				s.pop_back();
				board[i][j] = ch;
			}
		}
	}

	bool canProceed(string s, const vector<string>& words) {
		for (string word : words) {
			if (word.starts_with(s)) return true;
		}
		return false;
	}
};
