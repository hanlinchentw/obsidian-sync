class Solution {
public:
	vector<int> searchRange(vector<int>& nums, int target) {
		int left = 0, right = nums.size()-1;

		int lowerBound = -1;
		while (left <= right) {
			int mid = left + (right-left) / 2;
			if (nums[mid] == target)
				lowerBound = mid;

			if (nums[mid] >= target) {
				right = mid - 1;
			} else {
				left = mid + 1;
			}
		}

		left = 0, right = nums.size() - 1;
		int upperBound = -1;
		while (left <= right) {
			int mid = left + (right - left) / 2;
			if (nums[mid] == target)
				upperBound = mid;

			if (nums[mid] > target) {
				right = mid - 1;
			} else {
				left = mid + 1;
			}
		}

		return {lowerBound, upperBound};
	}
};
