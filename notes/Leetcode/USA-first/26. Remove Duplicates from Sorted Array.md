class Solution {
public:
	int removeDuplicates(vector<int>& nums) {
		unordered_set<int> seen;
		int k = 1;
		for (int i = 1; i < nums.size(); i++) {
			if (nums[i] == nums[k-1])
				continue;
			nums[k++] = nums[i];
		}
		return k;
	}
};
