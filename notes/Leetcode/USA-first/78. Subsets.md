```cpp
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> ans;
        vector<int> comb;
        dfs(nums, 0, comb, ans);
        return ans;
    }

	void dfs(const vector<int>& nums,
			int index,
			vector<int>& comb,
			vector<vector<int>>& ans) {
		ans.push_back(comb);
		for (int i = index; i < nums.size(); i++) {
			comb.push_back(nums[i]);
			dfs(nums, i + 1, comb, ans);
			comb.pop_back();
		}	
	}
};
```
