### Thinking：

1. 每找到一個 1 就馬上展開 BFS，經過的點設為 INT_MAX （避免再次訪問）
2. 每次 BFS 完  count++

### Solution:

```
class Solution {
public:
	int numIslands(vector<vector<char>>& grid) {
		int n = grid.size();
		int m = grid[0].size();

		int count = 0;

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (grid[i][j] == '1') {
					// do BFS
					queue<pair<int, int>> que;
					que.push({i, j});
					while (!que.empty()) {
						auto front = que.front(); que.pop();
						int r = front.first;
						int c = front.second;
						if (grid[r][c] != '1') 
							continue;
						// top
						if (r > 0) {
							que.push({r-1, c});
						}
						// bottom
						if (r < n - 1) {
							que.push({r+1, c});
						}
						// left
						if (c > 0) {
							que.push({r, c-1});
						}
						// right
						if (c < m - 1) {
							que.push({r, c+1});
						}
						grid[r][c] = '-1';
					}
					count++;
				}
			}
		}
		return count;
	}
};
```
