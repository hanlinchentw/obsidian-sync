class Solution {
public:
	int lengthOfLongestSubstring(string s) {
		unordered_set<char> set;
		int left = 0;
		for (int right = 0; right < s.length(); right++) {
			if (set.find(s[right]) != set.end()) {
				set.erase(s[right]);
				left++;
			} else {
				set.insert();
			}
		}
	}
};
