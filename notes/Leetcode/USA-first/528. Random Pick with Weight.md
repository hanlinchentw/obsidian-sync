class Solution {
private:
	vector<int> wSum;
	int total = 0;
public:
	Solution(vector<int>& w) {
		for (auto weight: w) {
			total += weight;
			wSum.push_back(total);
		}
	}

	int pickIndex() {
		int target = rand() % total + wSum[0];
		int left = 0;
		int right = wSum.size() - 1;
		while (left < right) {
			int mid = (left + right) >> 1;
			if (target > mid) {
				left = mid + 1;
			} else {
				right = mid;
			}
		}
		return left;
	}
};
