				  int target, 
				  int sum, 
				  vector<int>& curr,
				  set<vector<int>>& visited) {
		if (sum == target) {
			visited.insert(curr);
			return;
		}
		if (sum > target) return;
		for (int i = 0; i < nums.size(); i++) {
			canAddup(nums, target);
		}
	}
};
