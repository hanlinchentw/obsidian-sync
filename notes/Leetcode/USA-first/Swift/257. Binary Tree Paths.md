	func dfs(_ node: TreeNode?) -> [[Int]] {
		if node == nil { return [] }
		if node?.left == nil && node?.right == nil {
			return [[node!.val]]
		}
		let left = dfs(node?.left)
		let right = dfs(node?.right)
		var sumPath = left+right
		for index in 0 ..< sumPath.count {
			sumPath[0].insert(node!.val, at: 0)
		}
		return sumPath
	}
}
