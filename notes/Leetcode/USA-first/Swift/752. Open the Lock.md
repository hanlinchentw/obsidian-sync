	void dfs(const unordered_set<string>& deadends, const string& target, string curr, unordered_set<string>& visited, int layer) {
		if (deadends.count(curr) || visited.count(curr))
			return;
		if (curr == target) {
			ans = layer;
			return;
		}
		visited.insert(curr);

		string switchFirstUp = curr;
		int intValue = curr[0] - '0';
		switchFirstUp[0] = (intValue-1) + '0';
		dfs(deadends, target, switchFirstUp, visited, layer+1);

		string switchFirstDown = curr;
		intValue = curr[0] - '0';
		switchFirstDown[0] = (intValue+1) + '0';
		dfs(deadends, target, switchFirstDown, visited, layer+1);

		string switchSecondUp = curr;
		intValue = curr[1] - '0';
		switchSecondUp[1] = (intValue-1) + '0';
		dfs(deadends, target, switchSecondUp, visited, layer+1);

		string switchSecondDown = curr;
		intValue = curr[1] - '0';
		switchSecondDown[1] = (intValue+1) + '0';
		dfs(deadends, target, switchSecondDown, visited, layer+1);

		string switchThirdUp = curr;
		intValue = curr[2] - '0';
		switchThirdUp[2] = (intValue-1) + '0';
		dfs(deadends, target, switchThirdUp, visited, layer+1);

		string switchThirdDown = curr;
		intValue = curr[2] - '0';
		switchThirdDown[2] = (intValue+1) + '0';
		dfs(deadends, target, switchThirdDown, visited, layer+1);

		string switchFourUp = curr;
		intValue = curr[3] - '0';
		switchFourUp[3] = (intValue-1) + '0';
		dfs(deadends, target, switchFourUp, visited, layer+1);

		string switchFourDown = curr;
		intValue = curr[3] - '0';
		switchFourDown[3] = (intValue+1) + '0';
		dfs(deadends, target, switchFourDown, visited, layer+1);
	}
};
