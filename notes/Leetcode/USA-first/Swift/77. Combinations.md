class Solution {
    func combine(_ n: Int, _ k: Int) -> [[Int]] {
	    var ans = [[Int]]()
	    dfs(n, k, 0, [], &ans)
	    return ans
    }

	func dfs(_ n: Int, _ k: Int, _ curr: Int, _ comb: [Int], _ ans: inout [[Int]]) {
		if comb.count == k {
			ans.append(comb)
			return
		}
		if curr+1 > n { return }
		for i in curr+1...n {
			dfs(n, k, i, comb + [i], &ans)
		}
	}
}
