class Solution {
    func searchRange(_ nums: [Int], _ target: Int) -> [Int] {
        var left = 0, right = nums.count - 1
        var lower = -1
        while left <= right {
	        var mid = (left+right) / 2
	        if nums[mid] == target { lower = mid }
	        if nums[mid] >= target {
		        right = mid - 1
	        } else {
		        left = mid + 1
	        }
        }
		left = 0
		right = nums.count - 1
		var upper = -1
        while left <= right {
	        var mid = (left+right) / 2
	        if nums[mid] == target { upper = mid }
	        if nums[mid] <= target {
		        left = mid + 1
	        } else {
		        right = mid - 1
	        }
        }
        return [lower, upper]
    }
}
