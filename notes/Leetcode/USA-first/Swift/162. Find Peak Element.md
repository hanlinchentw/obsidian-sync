class Solution {
    func findPeakElement(_ nums: [Int]) -> Int {
        var left = 0, right = nums.count - 1
        while left < right {
	        let mid = (left + right) / 2
	        if nums[mid] < nums[mid+1] {
		        left = mid + 1
	        } else {
		        right = mid
	        }
        }
        return left
    }
}
