### Thinking：
1. 照最高階排序
2. 若最高階的數字一樣，則往下比較
3. 最後組合起來

### Solution:

```
class Solution {
    func largestNumber(_ nums: [Int]) -> String {
        let sortedNums = nums.sorted {
	        var first = Array(String($0))
	        var second = Array(String($1))

			while let left = first.first, let right = second.first {
				if left > right {
					return true
				} else if left < right {
					return false
				} else {
					if first.length > 1 { first.removeFirst() }
					if second.length > 1 { second.removeFirst() }
				}
			}
	        return true
        }
        var ans = String()
        for num in sortedNums {
	        ans += String(num)
        }
        return ans
    }
}
```
