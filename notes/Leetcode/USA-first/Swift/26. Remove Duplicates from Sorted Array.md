class Solution {
    func removeDuplicates(_ nums: inout [Int]) -> Int {
        var k = 0
        var map = [Int: Bool]()
        for i in nums.indices {
	        let num = nums[i]
	        if map[num] == nil {
		        nums.swapAt(i, k)
		        k += 1
		        map[num] = true
	        }
        }
        return k
    }
}
