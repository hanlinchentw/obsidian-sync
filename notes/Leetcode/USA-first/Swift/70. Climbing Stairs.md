class Solution {
    func climbStairs(_ n: Int) -> Int {
        var dp = Array(repeating: 0, count: n+1)
        dp[0] = 0
        dp[1] = 1
        dp[2] = 2
        var count = 3
        while count != n {
	        dp[count] = dp[count - 1]  + dp[count - 2]
	        count += 1
        }
        return dp[n]
    }
}
