class Solution {
    func sortColors(_ nums: inout [Int]) {
        var map = [Int:Int]()
        for num in nums {
	        map[num, default: 0] += 1
        }
        for index in nums.indices {
	        if map[0, default: 0] != 0 {
		        nums[index] = 0
		        map[0, default: 0] -= 1
	        } else if map[1, default: 0] != 0 {
		        nums[index] = 1
		        map[1, default: 0] -= 1
	        } else {
		        nums[index] = 2
		        map[2, default: 0] -= 1
	        }
        }
    }
}
