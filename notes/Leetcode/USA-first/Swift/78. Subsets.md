class Solution {
    func subsets(_ nums: [Int]) -> [[Int]] {
	    var ans = [[Int]]()
		backtrace(0, nums, [], &ans)
	    return ans
    }

	func backtrace(_ start: Int, _ nums: [Int], _ comb: [Int], _ ans: inout [[Int]]) {
		ans.append(comb)
		if start >= nums.count { return }
		for i in start ..< nums.count {
			backtrace(i+1, nums, comb + [nums[i]], &ans)
		}
	}
}
