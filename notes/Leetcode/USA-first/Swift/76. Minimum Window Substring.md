class Solution {
    func minWindow(_ s: String, _ t: String) -> String {
        var tMap = [Character: Int]()
        for char in Array(t) { tMap[char, default: 0]  += 1 }
        var have = 0, need = tMap.keys.count
        var left = 0
        var minStart = 0
        var minLen = Int.max
        var sArr = Array(s)
        for right in 0 ..< s.count {
	        let char = sArr
	        tMap[char, default: 0]  -= 1
	        if tMap[char, default: 0] == 0 {
		        have += 1
	        }
	        while have == need {
		        if right - left + 1 < minLen {
			        minLen = right - left + 1
			        minStart = left
		        }
		        
	        }
        }
        let end = minStart + minLen
        return minLen == Int.max ? "" : String(s[minStart..<end])
    }
}
