class Solution {
    func searchMatrix(_ matrix: [[Int]], _ target: Int) -> Bool {
	    let n = matrix.count, m = matrix[0].count
        var left = 0, right = n*m - 1
        while left <= right {
	        let mid = (left + right) / 2
	        let row = mid / m
	        let col = mid % m
	        if matrix[row][col] == target { return true }
	        if matrix[row][col] > target {
		        right = mid - 1
	        } else {
		        left = mid + 1
	        }
        }
        return false
    }
}
