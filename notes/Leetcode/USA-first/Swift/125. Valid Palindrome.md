class Solution {
    func isPalindrome(_ s: String) -> Bool {
        var left = 0, right = s.count - 1
        while left != right {
	        while s.char(left).isAlphabet
        }
    }
}

extension String {
	func char(at index: Int) -> Character {
		s[s.index(s.startIndex, offsetBy: index)]
	}
}
