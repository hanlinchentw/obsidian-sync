class Solution {
public:
    int diameterOfBinaryTree(TreeNode* root) {
        if (!root) return 0;
        int maximum = INT_MIN;
        dfs(root, maximum);
        return maximum;
    }

	int dfs(TreeNode* node, int& maximum) {
		if (!node) return 0;
		int left = dfs(node->left, maximum);
		int right = dfs(node->right, maximum);
		maximum = max(maximum, left+ right);
		return max(left, right) + 1;
	}
};
