```cpp
class MyStack {
private:
	queue<int> que;
public:
    MyStack() {
        
    }
    
    void push(int x) {
	    que.push(x);
    }
    
    int pop() {
        int size = que.size() - 1;
        while (size--) {
	        int num = que.front(); 
	        que.pop();
	        push(num);
        }
        int ans = que.front(); que.pop();
        return ans;
    }
    
    int top() {
        int size = que.size() - 1;
        while (size--) {
	        int num = que.front(); 
	        que.pop();
	        push(num);
        }
        int ans = que.front(); que.pop();
        push(ans);
        return ans;
    }
    
    bool empty() {
        return que.empty();
    }
};
```
