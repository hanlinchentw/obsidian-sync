class Solution {
public:
    vector<vector<int>> kClosest(vector<vector<int>>& points, int k) {
	    priority_queue<vector<int, int> q(points.begin(), points.begin(), [](const vector<int>& point1, const vector<int>& point2) {
		    int distance1 = (point1[0]**2 + point1[1]**2);
		    int distance2 = (point2[0]**2 + point2[1]**2);
		    return distance1 > distance2;
	    });
	    vector<vector<int>> ans;
	    while (k--) {
		    ans.push_back(q.front());
		    q.pop();
	    }
	    return ans;
    }
};
