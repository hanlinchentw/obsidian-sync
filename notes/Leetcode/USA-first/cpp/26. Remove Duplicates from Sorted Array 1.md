class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int k = 0;
        unordered_set<int> set;
        for (int i = 0; i < nums.size(); i++) {
	        if (set.count(nums[i]) == 0) {
		        set.insert(nums[i]);
		        swap(nums[i], nums[k++]);
	        }
        }
        return k;
    }
};
