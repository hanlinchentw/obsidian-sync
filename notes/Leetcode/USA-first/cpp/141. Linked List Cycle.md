class Solution {
public:
    bool hasCycle(ListNode *head) {
		unordered_set<ListNode*> set;
		ListNode* curr = head;
		while (curr) {
			if (set.count(curr)) return true;
			set.insert(curr);
			curr = curr->next;
		}
		return false;
    }
};

class Solution {
public:
    bool hasCycle(ListNode *head) {
	    if (!head || !head->next) return false;
	    ListNode* slow = head;
	    ListNode* fast = head;
	    while (fast->next && fast->next->next) {
		    slow = slow->next;
		    fast = fast->next->next;
		    if (slow == fast) return true;
	    }
	    return false;
    }
};
