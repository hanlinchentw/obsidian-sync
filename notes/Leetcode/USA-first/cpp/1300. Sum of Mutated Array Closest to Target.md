class Solution {
private:
	int diff(int val, vector<int>& arr, int target) {
		int sum = 0;
		for (auto num: arr) {
			sum += min(num, val);
		}
		return abs(sum - target);
	}
public:
	int findBestValue(vector<int>& arr, int target) {
		int left = 0, right = target;
		while (left < right) {
			int mid = (left + right) >> 1;
			if (diff(mid, arr, target) > diff(mid+1, arr, target)) {
				left = mid + 1;
			} else {
				right = mid;
			}
		}
		return left;
	}
};
