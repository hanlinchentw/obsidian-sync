class Solution {
public:
    int search(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;

		while (left <= right) {
			int mid = (left + right) / 2;
			if (nums[mid] == target) 
				return mid;
			if (nums[right] > nums[mid]) {
				if (nums[mid] < target && target <= nums[right]) {
					low = mid + 1;
				} else {
					high = mid - 1;
				}
			} else {
				if (nums[left] <= target && target <= nums[mid]) {
					high = mid - 1;
				} else {
					low = mid + 1;
				}
			}
		}
	
        return -1;
    }
};
