class Solution {
public:
    bool isPalindrome(string s) {
        int left = 0, right = s.size() - 1;
        while (left < right) {
	        while (left < right && !isdigit(s[left]) && !isalpha(s[left]))
		        left += 1;
		    while (left < right && !isdigit(s[right]) && !isalpha(s[right]))
		        right -= 1;
		    if (toupper(s[left]) != toupper(s[right]))
			    return false;
			left += 1;
			right -= 1;
        }
        return true;
    }
};
