### Solution:

```cpp
class Solution {
public:
    ListNode* oddEvenList(ListNode* head) {
	    if (!head || !head->next) return head;
        ListNode* evenHead = head->next;
        ListNode* even = evenHead;
        ListNode* odd = head;
        while (even && even->next) {
	        odd->next = even->next;
	        odd = odd->next;
	        even->next = odd->next;
	        even = even->next;
        }
        odd->next = evenHead;
        return head;
    }
};
```
