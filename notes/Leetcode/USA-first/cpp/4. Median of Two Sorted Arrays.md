### Solution:
Brute force O(m+n)
```cpp
class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
	    int m = nums1.size(), n = nums2.size();
		int m1 = 0, m2 = 0;
		int mid = (m + n) / 2;
		int i = 0, j = 0;
		for (int count = 0; count < mid; count++) {
			m2 = m1;
			if (i < n && j < m) {
				if (nums1[i] > nums2[j]) {
					m1 = nums2[j++];
				} else {
					m1 = nums1[i++];
				}
			} else if (i < n) {
				m1 = nums1[i++];
			} else {
				m1 = nums2[j++];
			}
		}
		if ((m+n % 2) != 0) {
			return (double)(m1 + m2) / 2;
		} else {
			return (double)m1;
		}
    }
};
```
