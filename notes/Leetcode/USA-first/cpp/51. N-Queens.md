class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<string> combs;
        queue<string> que;
        for (int i = 0; i < n; i++) {
	        que.push(to_string(i));
        }
        while (!que.empty()) {
	        int size = que.size();
	        while (size--) {
		        auto front = que.front(); que.pop();
		        if (front.size() == n) {
			        combs.push_back(front);
			        continue;
		        }
		        for (int i = 0; i < n; i++) {
			        if (isValid(n, front, i)) {
				        que.push(front + to_string(i));
			        }
		        }
	        }
        }

		vector<vector<string>> ans;
		for (auto comb : combs) {
			vector<string> pos;
			for (int i = 0; i < n; i++) {
				string p(n, '.');
				p[i] = 'Q';
				pos.push_back(p);
			}
			ans.push_back(pos);
		}
		return ans;
    }

	bool isValid(int n, string current, int newPos) {
		bool valid = true;
		int row = current.size();
		for (int i = 0; i < n; i++) {
			int intVal = current[i] - '0';
			if (intVal == newPos) { 
				valid = false;
				break;
			}
            int rowDiff = row - i;
			if (intVal + rowDiff == newPos || intVal - rowDiff == newPos) {
				valid = false;
				break;
			}
		}
		return valid;
	}
};
