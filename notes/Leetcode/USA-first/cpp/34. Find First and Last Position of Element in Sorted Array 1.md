class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int lowerBound = -1, upperBound = -1;
		int low = 0, high = nums.size() - 1;
		while (low <= high) {
			int mid = (low + high) / 2;
			if (nums[mid] == target) 
				lowerBound = mid;
			if (nums[mid] >= target) {
				high = mid - 1;
			} else {
				low = mid + 1;
			}
		}
		low = 0, high = nums.size() - 1;
		while (low <= high) {
			int mid = (low + high) / 2;
			if (nums[mid] == target)
				upperBound = mid;
			if (nums[mid] > target) {
				high = mid - 1;
			} else {
				low = mid + 1;
			}
		}
		return {lowerBound, upperBound};
    }
};
