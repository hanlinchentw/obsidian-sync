class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
	    let i = quickSelect(nums, 0, nums.size()-1);
		    if (i < k) {
			    quickSelect(nums, i+1, nums.size()-1);
		    } else if (i > k) {
			    quickSelect(nums, 0, i-1]);
		    } else {
			    return i;
		    }
    }

	int quickSelect(vector<int>& nums, int low, int high) {
		int p = high;
		int x = nums[high];
		int i = low;
		for (int j = 0; j < high; j++) {
			if (nums[j] < x) {
				swap(nums[i++], nums[j]);
			}
		}
		swap(nums[i], nums[p]);
		return i;
	}
};
