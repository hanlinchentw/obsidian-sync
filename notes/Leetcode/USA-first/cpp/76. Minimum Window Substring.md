class Solution {
public:
    string minWindow(string s, string t) {
	    if (t.size() > s.size()) return "";
        unordered_map<char, int> tMap;
        for (auto c : t) tMap[c]++;
        int need = tMap.size();
        string ans = s;
        int maxLen = INT_MAX
        int left = 0;
        int have = 0;
        for (int right = 0; right < s.size(); right++) {
	        char c = s[right];
			if (tMap.count(c) && tMap[c] > 0)
				have++;
			tMap[c]--;
	        while (have == need) {
		        if (right - left + 1 < maxLen) {
			        maxLen = min(maxLen, right - left + 1);
			        ans = s.substr(left, right-left+1);
		        }
		        char leftChar = s[left];
			    if (tMap.count(leftChar) && tMap[leftChar] == 0) {
				    tMap[leftChar]++;
				    have--;
			    }
				left++;
	        }
        }
        return maxLen == INT_MAX ? "" : ans;
    }
};
