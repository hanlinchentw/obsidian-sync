class Solution {
    func reverseStr(_ s: String, _ k: Int) -> String {
	    var sArr = Array(s)
	    for i in stride(from: 0, to: s.count-1, by: 2*k) {
		    var left = i, right = i+k-1 >= s.count ? s.count-1 : i+k-1
		    while (left < right) {
			    sArr.swapAt(left, right)
			    left += 1
			    right -= 1
		    }
        }
        return String(sArr)
    }
}
