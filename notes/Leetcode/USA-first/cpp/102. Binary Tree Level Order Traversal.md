BFS - Iterative
```
class Solution {
public:
	vector<vector<int>> levelOrder(TreeNode* root) {
		if (root == NULL) return {};
		vector<vector<int>> ans;
		queue<TreeNode*> que;
		que.push(root);

		while (!que.empty()) {
			int size = que.size();
			vector<int> level;
			while (size--) {
				TreeNode* front = que.front(); que.pop();	
				if (front->left) que.push(front->left);
				if (front->right) que.push(front->right);
				level.push_back(front->val);
			}
			ans.push_back(level);
		}
		return ans;
	}
};

BFS - recursive
```
class Solution {
public:
	vector<vector<int>> levelOrder(TreeNode* root) {
		if (root == NULL) return {};
		vector<vector<int>> ans;
		traversal(root, 0, ans);
		return ans;
	}

	void traversal(TreeNode* node, int level, vector<vector<int>>& ans) {
		if (node == NULL) return;
		if (ans.empty() || level >= ans.size()) {
			ans.push_back({node->val});
		} else {
			ans[level].push_back(node->val);
		}
		traversal(node->left, level+1, ans);
		traversal(node->right, level+1, ans);
	}
};
