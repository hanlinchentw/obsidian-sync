class Solution {
public:
	int mySqrt(int x) {
		if (x == 0 || x == 1) return x;
		int left = 1, right = x;
		while (left <= right) {
			int mid = (left + right) / 2;
			long square = static_cast<long>(mid) * mid;
			if (square == x)
				return mid;

			if (square > x) {
				right = mid - 1;
			} else {
				left = mid + 1;
			}
		}
		return right;
	}
};
