### Solution:

Min heap

```cpp
class Solution {
public:
	int kthSmallest(vector<vector<int>>& matrix, int k) {
		priority_queue<
			int,
			vector<int>,
			greater<int>
	    > que;
	    for (int i = 0; i < matrix.size(); i++) {
			for (int j = 0; j < matrix[0].size(); j++) {
				que.push(matrix[i][j]);
			}
	    }
		int ans = 0;
		while (k--) {
			ans = que.top(); que.pop();
		}
		return ans;
	}
};
```

Binary search
```c++
class Solution {
public:
	int kthSmallest(vector<vector<int>>& matrix, int k) {
		int n = matrix.size(), m = matrix[0].size();
		int lo = matrix[0][0], hi = matrix[n-1][m-1];
		while (lo < hi) {
			int mid = (lo + hi) >> 1;
			int count = 0, j = m - 1;
			for (int i = 0; i < n; i++) {
				while (j >= 0 && matrix[i][j] > mid) j--;
				count += j+1;
			}
			if (count < k) lo = mid + 1;
			else hi = mid;
		}
		return lo;
	}
};
```
