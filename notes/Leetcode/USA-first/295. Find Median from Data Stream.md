class MedianFinder {
private:
	priority_queue<int, vector<int>, greater<>> que;
	priority_queue<int, vector<int>, greater<>> tmpQue;
public:
	MedianFinder() {
	}

	void addNum(int num) {
		que.push(num);
	}

	double findMedian() {
		double median;
		if (que.size() % 2 == 0) {
			int idx = que.size() / 2;
			int top;
			while (idx--) {
				top = que.top(); que.pop();
				tmpQue.push(top);
			}
			median = (double)(top + que.top()) / 2;
		} else {
			int idx = (que.size() / 2) + 1;
			int top;
			while (idx--) {
				top = que.top(); que.pop();
				tmpQue.push(top);
			}
			median = top;
		}
		while (!que.empty()) {
			tmpQue.push(que.top()); que.pop();
		}
		que = tmpQue;
		tmpQue = priority_queue<int, vector<int>, greater<>>();
		return median;
	}
};

class MedianFinder {
private:
	priority_queue<int> maxHeap; // store lower numbers
	priority_queue<int, vector<int>, greater<>> minHeap; // store higher numbers
public:
	MedianFinder() {}
	void addNum(int num) {
		maxHeap.push(num);
		minHeap.push(maxHeap.top());
		maxHeap.pop();
		if (minHeap.size() > maxHeap.size()) {
			maxHeap.push(minHeap.top());
			minHeap.pop();
		}
	}

	double findMedian() {
		if (maxHeap.size() > minHeap.size()) return maxHeap.top();
		return (maxHeap.top() + minHeap.top()) / 2.0 ;
	}
};
