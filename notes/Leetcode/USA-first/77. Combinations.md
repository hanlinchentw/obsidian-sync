### Solution:

one recurssion
Time complexity should be O(n*k)

```cpp
class Solution {
public:
    vector<vector<int>> combine(int n, int k) {
	    vector<vector<int>> ans;
	    dfs(n, k, 1, {}, ans);
        return ans;
    }

	void dfs(int n, int k, int start, vector<int> comb, vector<vector<int>>& ans) {
		if (comb.size() == k) {
			ans.push_back(comb);
			return;
		}
		for (int i = start; i <= n; i++) {
			comb.push_back(i);
			dfs(n, k, i+1, comb, ans);
			comb.pop_back();
		}	
	}
};
```
