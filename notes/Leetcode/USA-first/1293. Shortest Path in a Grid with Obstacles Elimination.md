class Solution {
public:
	int shortestPath(vector<vector<int>>& grid, int k) {
		int n = grid.size();
		int m = grid[0].size();
		queue<pair<int, pair<int, pair<int, int>>>> que; // remainK, count, point
		que.push({k, {1, {0, 0}}});
		set<pair<int, int>> visited;
		while (!que.empty()) {
			auto [remainK, p] = que.front(); que.pop();
			int count = p.first;
			auto point = p.second;
			int row = point.first;
			int col = point.second;
			if (row == n-1 && col == m-1)
				return count;
			if (visited.count(point))
				continue;
			vector<pair<int, int>> dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
			for (auto dir : dirs) {
				int nr = row + dir.first;
				int nc = col + dir.second;
				if (0 <= nr && nr < n && 0 <= nc && nc < m) {
					if (grid[row][col] == 0) {
						que.push({remainK, {count+1, {nr, nc}}});					
					} else if (remainK > 0) {
						que.push({remainK-1, {count + 1, {nr, nc}}});
					}
				}
			}
		}
		return -1;
	}
};
