### Solution:

find non-zero
```
class Solution {
public:
	void moveZeroes(vector<int>& nums) {
		int j = 0;
		for (int i = 0; i < nums.size(); i++) {
			if (nums[i] != 0) {
				swap(nums[i], nums[j]);
				j++;
			}
		}
	}
};
```

void moveZeroes(vector<int>& nums) {
	int insertPos = 0;
	for (int i = 0; i < nums.size(); i++) {
		if (nums[i] != 0) {
			swap(nums[insertPos++], nums[i]);
		}
	}
}
