index % 2 == 0 == (index+1) % 2 == 1
### Solution:

```
class Solution {
public:
	int singleNonDuplicate(vector<int>& nums) {
		int left = 0, right = nums.size() - 1;

		while (left < right) {
			int mid = (left + right) >> 1;
			if (mid % 2 == 0) {
				if (nums[mid] != nums[mid + 1]) {
					right = mid;
				} else {
					left = mid + 1;
				}
			} else {
				if (nums[mid] != nums[mid - 1]) {
					right = mid;
				} else {
					left = mid + 1;
				}
			}
		}
		return left;
	}
};
```
